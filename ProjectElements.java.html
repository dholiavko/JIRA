import org.openqa.selenium.WebDriver; 
import org.openqa.selenium.WebElement; 
import org.openqa.selenium.support.FindBy; 
import org.openqa.selenium.support.How; 
 
import java.util.List; 
 
public class ProjectElements { 
 
    protected WebDriver webDriver; 
 
    public ProjectElements(WebDriver webDriver) { 
        this.webDriver = webDriver; 
    } 
 
    @FindBy (xpath="//table[@class='aui']/tbody/tr") 
    private List<WebElement> project_table; 
 
    public List<WebElement> getProject_table() { 
        return project_table; 
    } 
 
    @FindBy (how = How.XPATH, using = "//table[@class='aui']/tbody/tr/td[2]/a") 
    private List<WebElement> project_list; 
 
    public List<WebElement> getProject_list () { 
        return project_list; 
    } 
 
}
