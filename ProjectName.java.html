import org.openqa.selenium.By; 
import org.openqa.selenium.WebDriver; 
import org.openqa.selenium.WebElement; 
import org.openqa.selenium.support.PageFactory; 
import org.openqa.selenium.support.ui.ExpectedConditions; 
import org.openqa.selenium.support.ui.WebDriverWait; 
import java.util.List; 
import java.util.concurrent.TimeUnit; 
 
public class ProjectName { 
 
    protected static ProjectElements pe; 
    protected static ProjectFields pf; 
 
    private WebDriver webDriver; 
    public ProjectName(WebDriver webDriver) { 
        this.webDriver = webDriver; 
    } 
 
    public void ProjectNameTest() { 
 
        pf = new ProjectFields(webDriver); 
        pe = PageFactory.initElements(webDriver, ProjectElements.class); 
 
//        List<WebElement> Projects_list = webDriver.findElements(By.xpath("//table[@class='aui']/tbody/tr")); 
//        int list_size = Projects_list.size(); 
 
        int list_size = pe.getProject_table().size(); 
 
        System.out.println(list_size); 
        int current_link; 
        int n = 1; 
        for (current_link = 0; current_link < list_size; current_link++) { 
            System.out.println("----------------------------------------------------------------"); 
 
//            Projects_list = webDriver.findElements(By.xpath("//table[@class='aui']/tbody/tr/td[2]/a")); 
            //             Projects_list = webDriver.findElements(By.xpath("//table[@class='aui']/tbody/tr[323]/td[2]/a")); 
//            WebElement el = Projects_list.get(current_link); 
 
            WebElement el = pe.getProject_list().get(current_link); 
            System.out.println(n + " Project Name: " + el.getText()); 
            el.click(); 
            // NOTE: Verify presence of components. 
            WebElement Component = null; 
            try { 
                Component = (new WebDriverWait(webDriver, 3).until(ExpectedConditions.presenceOfElementLocated(By 
                                .xpath("//div[@id='project-config-webpanel-summary-components']/div[2]/div/p/a[text()='Add a component']")))); 
            } catch (Exception e) { 
            } 
            if (Component != null) { 
                System.out.println("NO COMPONENT FOUND"); 
                pf.Roles(); 
                webDriver.navigate().back(); 
                pf.Workflow(); 
                webDriver.navigate().back(); 
            } else { 
                WebElement Show_all = null; 
                try { 
                    Show_all = (new WebDriverWait(webDriver, 3).until(ExpectedConditions.presenceOfElementLocated(By 
                            .xpath("//div[@id='project-config-webpanel-summary-components']/div/div/p/*[text()='Show all']")))); 
                } catch (Exception e) { 
                } 
                if (Show_all != null) { 
 
                    WebElement Show_all_Button = (new WebDriverWait(webDriver, 3).until(ExpectedConditions.presenceOfElementLocated(By 
                            .xpath("//div[@id='project-config-webpanel-summary-components']/div/div/p/*[text()='Show all']")))); 
                    Show_all_Button.click(); 
 
                    List<WebElement> Show_all_Components = (new WebDriverWait(webDriver, 3).until(ExpectedConditions.presenceOfAllElementsLocatedBy(By 
                            .xpath("//table[@id='project-config-components-table']/tbody[2]//tr")))); 
                    int Number_of_Components = Show_all_Components.size(); 
                    System.out.println("SIX OR MORE COMPONENTS: " + Number_of_Components); 
                    for (WebElement sac : Show_all_Components) { 
                        System.out.println( 
                                sac.findElement(By.xpath("td[2]")).getText() 
//                              + sac.findElement(By.xpath("td[3]")).getText() + " - " 
//                              + sac.findElement(By.xpath("td[4]")).getText() + " - " 
//                              + sac.findElement(By.xpath("td[5]")).getText() 
                        ); 
                    } 
                    webDriver.navigate().back(); 
                } else { 
                    pf.FiveOrLessComponentsSize(); 
                    System.out.println("FIVE OR LESS COMPONENTS: " + ProjectFields.Components_list_size); 
                    pf.FiveOrLessComponentsList(); 
                } 
             pf.Roles(); 
             webDriver.navigate().back(); 
             pf.Workflow(); 
             webDriver.navigate().back(); 
            } 
            webDriver.manage().timeouts().implicitlyWait(5, TimeUnit.SECONDS); 
            n++; 
        } 
    } 
}
