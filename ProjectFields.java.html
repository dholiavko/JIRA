import org.openqa.selenium.By; 
import org.openqa.selenium.WebDriver; 
import org.openqa.selenium.WebElement; 
import org.openqa.selenium.firefox.FirefoxDriver; 
import org.openqa.selenium.support.ui.ExpectedConditions; 
import org.openqa.selenium.support.ui.WebDriverWait; 
 
import java.util.List; 
 
public class ProjectFields { 
 
        private final WebDriver webDriver; 
        public ProjectFields(WebDriver webDriver) { 
        this.webDriver = webDriver; 
    } 
    public static int Components_list_size; 
 
    protected void Roles () { 
        WebElement ViewProjectRolesButton = (new WebDriverWait(webDriver, 3)) 
                .until(ExpectedConditions.presenceOfElementLocated(By.xpath("//*[@id='project-config-summary-people-list']/li[3]/span[2]/a"))); 
        ViewProjectRolesButton.click(); 
 
        WebElement PRoles = null; 
         try { 
             PRoles = (new WebDriverWait(webDriver, 3) 
                      .until(ExpectedConditions.presenceOfElementLocated(By.xpath("//table[@class='aui']/tbody/tr")))); 
          } catch (Exception e) { 
          } 
             if (PRoles != null) { 
                List<WebElement> Proj_Roles = (new WebDriverWait(webDriver, 3) 
                        .until(ExpectedConditions.presenceOfAllElementsLocatedBy(By.xpath("//table[@class='aui']/tbody/tr")))); 
//                List<WebElement> Proj_Roles = webDriver.findElements(By.xpath("//table[@class='aui']/tbody/tr")); 
                int Proj_Roles_Size = Proj_Roles.size(); 
                System.out.println("PROJECT ROLES: " + Proj_Roles_Size); 
 
                for (WebElement pr : Proj_Roles) { 
                    System.out.println( 
                        pr.findElement(By.xpath("td[1]/span")).getText() 
//                              + pr.findElement(By.xpath("td[2]")).getText() + " - " 
//                              + pr.findElement(By.xpath("td[3]")).getText() + " - " 
//                              + pr.findElement(By.xpath("td[4]")).getText() + " - " 
//                              + pr.findElement(By.xpath("td[5]")).getText() 
            ); 
        } 
        } 
             else { 
                    System.out.println("NO ROLES FOUND"); 
             } 
    } 
 
    protected void Workflow () { 
 
        List<WebElement> Workflow_Scheme = (new WebDriverWait(webDriver, 3) 
                .until(ExpectedConditions.presenceOfAllElementsLocatedBy(By.xpath("//ul/li/span/a[@class='project-config-workflow-name']")))); 
        int wfs = Workflow_Scheme.size(); 
        System.out.println("WORKFLOW SCHEME: " + wfs); 
        for (WebElement ws : Workflow_Scheme) { 
            System.out.println(ws.getText() + ";"); 
        } 
    } 
 
    protected void FiveOrLessComponentsSize () { 
        List<WebElement> Project_Components = (new WebDriverWait(webDriver, 3).until(ExpectedConditions.presenceOfAllElementsLocatedBy(By 
                .xpath("//ul[@id='project-config-summary-components-list']/li/span")))); 
        Components_list_size = Project_Components.size(); 
    } 
 
    protected void FiveOrLessComponentsList () { 
 
                List<WebElement> Project_Components = (new WebDriverWait(webDriver, 3) 
                        .until(ExpectedConditions.presenceOfAllElementsLocatedBy(By.xpath("//ul[@id='project-config-summary-components-list']/li")))); 
                Components_list_size = Project_Components.size(); 
                for (WebElement pc : Project_Components) { 
                    System.out.printf(pc.getText() + "%n"); 
                } 
    } 
}
