        webDriver.navigate().to(HomePage); 
 
        login = PageFactory.initElements(webDriver, JIRA2toolLinks.class);

        login.EnterLoginPassword(password); 
        login.ClickLoginButton(); 
        login.OpenAdminMenu(); 
        login.OpenAdminProjectMenu(); 
        login.ClickPCbutton();

    }
