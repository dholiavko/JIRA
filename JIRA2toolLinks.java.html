JIRA2toolLinks.java 
import org.openqa.selenium.WebDriver; 
import org.openqa.selenium.WebElement; 
import org.openqa.selenium.support.FindBy; 
import org.openqa.selenium.support.ui.ExpectedConditions; 
import org.openqa.selenium.support.ui.WebDriverWait; 
 
import java.util.ArrayList; 
import java.util.List; 
import java.util.concurrent.TimeUnit; 
 
public class JIRA2toolLinks { 
 
    protected WebDriver webDriver; 
 
    public JIRA2toolLinks(WebDriver webDriver) { 
        this.webDriver = webDriver; 
    } 
 
    @FindBy (id="login-form-username") 
    private WebElement LoginName; 
 
    @FindBy (id="login-form-password") 
    private WebElement LoginPassword; 
 
    @FindBy (id="login") 
    private WebElement LoginButton; 
 
    @FindBy (xpath="//div[@class='aui-header-primary']/ul/li[3]") 
    private WebElement IssuesButton; 
 
    @FindBy (xpath="//li[@id='issues_new_search_link']/a") 
    private WebElement Search_for_issues_Button; 
 
    @FindBy (id="admin_menu") 
//    @FindBy (xpath="//div[@class='aui-header-secondary']/ul/li[3]/a") 
    private WebElement AdminMenu; 
 
    @FindBy (id="admin_project_menu") 
    private WebElement AdminProjectMenu; 
 
    @FindBy (id="view_categories") 
    private WebElement PC_button; 
 
    @FindBy (xpath="//input[@id='login-form-authenticatePassword']") 
    private WebElement Confirm_login; 
 
    @FindBy (id="login-form-submit") 
    private WebElement Confirm_Login_Button; 
 
    @FindBy (xpath = "//table[@class='aui aui-table-rowhover']/tbody/tr[1]/td[3]/a") 
    protected List<WebElement> Customer_projects; 
 
    @FindBy (id="advanced-search") 
    protected WebElement Advanced_search; 
 
    @FindBy (xpath="//div[@class='search-options-container']/button") 
    protected WebElement Advanced_Search_Button; 
 
 
    public void EnterLoginName (String userLogin) { 
        ExpectedConditions.elementToBeClickable(LoginName); 
        LoginName.sendKeys(userLogin); 
    } 
    public void EnterLoginPassword (String userPassword) { 
        ExpectedConditions.elementToBeClickable(LoginPassword); 
        LoginPassword.sendKeys(userPassword); 
    } 
    public void ClickLoginButton() { 
        ExpectedConditions.elementToBeClickable(LoginButton); 
        LoginButton.click(); 
    } 
    public void OpenAdminMenu () { 
        ExpectedConditions.elementToBeClickable(AdminMenu); 
        AdminMenu.click(); 
    } 
    public void ClickIssuesButton() { 
        IssuesButton.click(); 
    } 
    public void ClickSearchButton() { 
        ExpectedConditions.elementToBeClickable(Search_for_issues_Button); 
        Search_for_issues_Button.click(); 
    } 
    public void OpenAdminProjectMenu () { 
        AdminProjectMenu.click(); 
    } 
    public void ClickPCbutton () { 
        ExpectedConditions.visibilityOf(PC_button); 
        PC_button.click(); 
    } 
    public void EnterConfirmLogin (String userPassword) { 
        Confirm_login.sendKeys(userPassword); 
    } 
    public void ClickConfirmLogin () { 
        Confirm_Login_Button.click(); 
    } 
    public void EnterAdvancedSearch (String project_search) { 
        Advanced_search.sendKeys(project_search); 
    } 
    public void ClickAdvancedSearchButton () { 
        Advanced_Search_Button.click(); 
    } 
 
    public void ClearAdvancedSearch () { 
        Advanced_search.clear(); 
    } 
 
    public List<WebElement> getCustomer_projects() { 
        return Customer_projects; 
    } 
 
 
}
