import org.openqa.selenium.WebDriver; 
import org.openqa.selenium.WebElement; 
import org.openqa.selenium.support.FindBy; 
 
public class LoginJIRAseven { 
 
    protected WebDriver webDriver; 
 
    public LoginJIRAseven(WebDriver webDriver) { 
        this.webDriver = webDriver; 
    } 
 
    @FindBy(id="login-form-username") 
    private WebElement LoginName; 
 
    @FindBy(id="login-form-password") 
    private WebElement LoginPassword; 
 
    @FindBy(id="login") 
    private WebElement LoginButton; 
 
    public void EnterLoginName (String userLogin) { 
        LoginName.sendKeys(userLogin); 
    } 
    public void EnterLoginPassword (String userPassword) { 
        LoginPassword.sendKeys(userPassword); 
    } 
    public void ClickLoginButton () { 
        LoginButton.click(); 
    } 
 
}
